### Let and Const

Using `var`

```js
function test() {
  foo = 2;

  for (var i = 0; i < 5; ++i) {
    var foo = i;
  }
}

test();
console.log(foo); // foo is not defined
```

But if you using `let`

```js
function test() {
  foo = 2;

  for (var i = 0; i < 5; ++i) {
    let foo = i;
  }
}

test();
console.log(foo); // 2
```

Using `const`

```js
var foo = 3;
foo = ['abc', 'def']; // foo is now bound to an array object

const bar = 3;
bar = ['abc', 'def']; // TypeError: assignment to const variable
```

### Generators, iterators, iterables and for...of

#### Generators

```js
function* Counter() {
  let i = 0;

  while (true) {
    const reset = yield i++;
    if (reset) {
      i = 0;
    }
  }
}

const gen = Counter();
console.log(gen.next().value); // 0
console.log(gen.next().value); // 1
console.log(gen.next().value); // 2
console.log(gen.next(true).value); // 0
console.log(gen.next().value); // 1
console.log(gen.next().value); // 2
```

#### Iterators

```js
function arrayIterator(arr) {
  var i = 0;
  return {
    next: function() {
      return arr[i++];
    },
    ended: i >= arr.length,
    reset: function() {
      i = 0;
    }
  };
}

var data = [0, 1, 2, 3, 4];
var iter = arrayIterator(data);
console.log(iter.next()); // 0
console.log(iter.next()); // 1
console.log(iter.next()); // 2
```

### Spread Syntax


### Destructuring

```js
let arr = [1, 2, 3, 4, 5];
let [a, b, ...rest] = arr;
console.log(a); // 1
console.log(b); // 2
console.log(rest); // [3, 4, 5]
```
